007import java.io.IOException;
008import java.util.zip.DataFormatException;
009import java.util.zip.Deflater;
010import java.util.zip.Inflater;
011
012public class CompressionZlib implements Compression {
013
014        public byte[] compress(byte[] in) {
015                int len = in.length * 2;
016                byte[] result = null;
017                while (result == null) {
018                        result = tryCompress(in, len);
019                        len *= 2;
020                }
021                return result;
022        }
023
024        private byte[] tryCompress(byte[] in, final int len) {
025                // Compress the bytes
026                final Deflater compresser = new Deflater(9, true);
027                compresser.setInput(in);
028                compresser.finish();
029
030                final byte[] output = new byte[len];
031                final int compressedDataLength = compresser.deflate(output);
032                if (compresser.finished() == false) {
033                        return null;
034                }
035                return copyArray(output, compressedDataLength);
036        }
037
038        public byte[] decompress(byte[] in) throws IOException {
039
040                final byte in2[] = new byte[in.length + 256];
041                for (int i = 0; i < in.length; i++) {
042                        in2[i] = in[i];
043                }
044
045                int len = in.length * 5;
046                byte[] result = null;
047                while (result == null) {
048                        result = tryDecompress(in2, len);
049                        len *= 2;
050                }
051                return result;
052        }
053
054        private byte[] tryDecompress(byte[] in, final int len) throws IOException {
055                // Decompress the bytes
056                final byte[] tmp = new byte[len];
057                final Inflater decompresser = new Inflater(true);
058                decompresser.setInput(in);
059                try {
060                        final int resultLength = decompresser.inflate(tmp);
061                        if (decompresser.finished() == false) {
062                                return null;
063                        }
064                        decompresser.end();
065
066                        return copyArray(tmp, resultLength);
067                } catch (DataFormatException e) {
068                        // e.printStackTrace();
069                        throw new IOException(e.toString());
070                }
071        }
072
073        private byte[] copyArray(final byte[] data, final int len) {
074                final byte[] result = new byte[len];
075                for (int i = 0; i < result.length; i++) {
076                        result[i] = data[i];
077                }
078                return result;
079        }
080
081}
