001package com.nafiux.grails.classdomainuml;
002
003/**
004 *
005 * @author nafiux
006 */
007public class AsciiEncoder implements URLEncoder {
008
009        final private char encode6bit[] = new char[64];
010        final private byte decode6bit[] = new byte[128];
011
012        public AsciiEncoder() {
013                for (byte b = 0; b < 64; b++) {
014                        encode6bit[b] = encode6bit(b);
015                        decode6bit[encode6bit[b]] = b;
016                }
017        }
018
019        public String encode(byte data[]) {
020                final StringBuilder resu = new StringBuilder((data.length * 4 + 2) / 3);
021                for (int i = 0; i < data.length; i += 3) {
022                        append3bytes(resu, data[i] & 0xFF, i + 1 < data.length ? data[i + 1] & 0xFF : 0,
023                                        i + 2 < data.length ? data[i + 2] & 0xFF : 0);
024                }
025                return resu.toString();
026        }
027
028        public byte[] decode(String s) {
029                if (s.length() % 4 != 0) {
030                        throw new IllegalArgumentException("Cannot decode " + s);
031                }
032                final byte data[] = new byte[(s.length() * 3 + 3) / 4];
033                int pos = 0;
034                for (int i = 0; i < s.length(); i += 4) {
035                        decode3bytes(data, pos, s.charAt(i), s.charAt(i + 1), s.charAt(i + 2), s.charAt(i + 3));
036                        pos += 3;
037                }
038                return data;
039        }
040
041        public int decode6bit(char c) {
042                return decode6bit[c];
043        }
044
045        public char encode6bit(byte b) {
046                assert b >= 0 && b < 64;
047                if (b < 10) {
048                        return (char) ('0' + b);
049                }
050                b -= 10;
051                if (b < 26) {
052                        return (char) ('A' + b);
053                }
054                b -= 26;
055                if (b < 26) {
056                        return (char) ('a' + b);
057                }
058                b -= 26;
059                if (b == 0) {
060                        return '-';
061                }
062                if (b == 1) {
063                        return '_';
064                }
065                assert false;
066                return '?';
067        }
068
069        private void append3bytes(StringBuilder sb, int b1, int b2, int b3) {
070                final int c1 = b1 >> 2;
071                final int c2 = ((b1 & 0x3) << 4) | (b2 >> 4);
072                final int c3 = ((b2 & 0xF) << 2) | (b3 >> 6);
073                final int c4 = b3 & 0x3F;
074                sb.append(encode6bit[c1 & 0x3F]);
075                sb.append(encode6bit[c2 & 0x3F]);
076                sb.append(encode6bit[c3 & 0x3F]);
077                sb.append(encode6bit[c4 & 0x3F]);
078        }
079
080        private void decode3bytes(byte r[], int pos, char cc1, char cc2, char cc3, char cc4) {
081                final int c1 = decode6bit[cc1];
082                final int c2 = decode6bit[cc2];
083                final int c3 = decode6bit[cc3];
084                final int c4 = decode6bit[cc4];
085                r[pos] = (byte) ((c1 << 2) | (c2 >> 4));
086                r[pos + 1] = (byte) (((c2 & 0x0F) << 4) | (c3 >> 2));
087                r[pos + 2] = (byte) (((c3 & 0x3) << 6) | c4);
088        }
089
090}
